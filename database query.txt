CREATE PROCEDURE [dbo].[GetPurchaseReport]  
    @PoNumber INT,  
    @startDate DATE,  
    @endDate DATE,  
    @supID INT,
	@BranchID varchar(50)
AS  
BEGIN  
    SELECT  
        pr.PurchaseID,  
        pr.PoNumber,  
        pr.PoReference,  
        pr.PoDate,  
        sp.SupplierName,  
        pm.PaymentMethodName,  
        pr.TotalAmount,  
        pr.TaxAmount,  
        pr.TotalAmount + pr.TaxAmount AS GrandTotal,  
        pr.PayableAmount,  
        pr.Balance  
    FROM  
        Purchase pr  
        INNER JOIN Suppliers sp ON pr.SupplierID = sp.SupplierID  
        INNER JOIN PaymentMethod pm ON pr.PaymentMode = pm.PaymentMethodID  
    WHERE  
        (pr.PurchaseID = @PoNumber  
        OR pr.PoDate BETWEEN @endDate AND @startDate  
        OR pr.SupplierID = @supID) and pr.BranchID = @BranchID 
END

CREATE PROCEDURE [dbo].[GetStockReportReport]
    @ProductID VARCHAR(50),
	@branchid varchar(50)
AS
BEGIN
    IF @ProductID IS NOT NULL
    BEGIN
        SELECT p.ProductName, pc.CategoryName, SUM(pd.Quantity) AS Quantity
        FROM PurchaseDetails pd
        INNER JOIN Products p ON p.ProductID = pd.ProductID
        INNER JOIN ProductCategory pc ON pc.ProductCategoryID = p.ProdcutCategoryID
        INNER JOIN SalesDetails s ON s.ProductID != p.ProductID
        WHERE (p.BranchID = @branchid) and (p.ProductID = @ProductID)
        GROUP BY p.ProductName, pc.CategoryName;
    END
    ELSE
    BEGIN
        SELECT p.ProductName, pc.CategoryName, SUM(pd.Quantity) AS Quantity
        FROM PurchaseDetails pd
        INNER JOIN Products p ON p.ProductID = pd.ProductID
        INNER JOIN ProductCategory pc ON pc.ProductCategoryID = p.ProdcutCategoryID
		INNER JOIN SalesDetails s ON s.ProductID != p.ProductID
		WHERE p.BranchID = @branchid
        GROUP BY p.ProductName, pc.CategoryName,p.ProductID;
    END
END
